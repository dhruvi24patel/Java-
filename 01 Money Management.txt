import java.util.Scanner;
public class Main 
{
    public static void main(String args[]) 
    {
      Scanner in =new Scanner(System.in);    
      System.out.print("Enter your income for this month: ");
      int x=in.nextInt();
      System.out.println(x);
      
      double nec=0.55;
      nec=nec*x;
      double ffa=0.1;
      ffa=ffa*x;
      double edu=0.1;
      edu=edu*x;
      double ltss=0.1;
      ltss=ltss*x;
      double play=0.1;
      play=play*x;
      double give=0.05;
      give=give*x;
      System.out.println(" NEC (Necessity): "+(int)nec);
      System.out.println("FFA (Finance Freedom Account) : "+(int)ffa);
      System.out.println("EDU (Education) : "+(int)edu);
      System.out.println("LTSS (Long Term Spending Saving) : "+(int)ltss);
      System.out.println("PLAY: "+(int)play);
      System.out.println("GIVE: "+(int)give);
    }
}
